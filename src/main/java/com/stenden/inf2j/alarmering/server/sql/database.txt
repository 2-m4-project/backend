CREATE TABLE public.client
(
id SERIAL PRIMARY KEY NOT NULL,
imei VARCHAR NOT NULL,
voornaam VARCHAR NOT NULL,
achternaam VARCHAR NOT NULL
)


CREATE TABLE public.geschiedenis
(
id SERIAL PRIMARY KEY NOT NULL,
client_id INT NOT NULL,
lat VARCHAR NOT NULL,
long VARCHAR NOT NULL,
tijd DATE,
melding VARCHAR NOT NULL
)

CREATE TABLE public.mantelzorger
(
id SERIAL NOT NULL,
voornaam VARCHAR NOT NULL,
achternaam VARCHAR NOT NULL,
email VARCHAR NOT NULL,
telefoonnummer VARCHAR NOT NULL,
gebruikersnaam VARCHAR NOT NULL,
wachtwoord VARCHAR NOT NULL
)

CREATE TABLE public.nieuws
(
id SERIAL PRIMARY KEY NOT NULL,
afbeelding VARCHAR NOT NULL,
text VARCHAR NOT NULL
)
